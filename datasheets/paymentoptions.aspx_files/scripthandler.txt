// using page id: billview/printpage


			var vz_included_elements = new Array();	// remember everything we've added to the DOM
													// this will redefine the existing array

			function vz_include_script( filename, defer )
			{
				var html_doc = document.getElementsByTagName('head').item(0);
				var js = document.createElement('script');
				js.setAttribute('language', 'javascript' );
				js.setAttribute('type', 'text/javascript');
				js.setAttribute('src', filename );
				if( defer )
					js.setAttribute('defer', 'defer');
				html_doc.appendChild(js);
				vz_included_elements.push(js);		
				return false;
			}

			function vz_include_image( filename )
			{
				var img = document.createElement('img');
				img.setAttribute('src', filename);
				document.body.appendChild(img);
				vz_included_elements.push(img);
				return false;
			}

			function vz_include_html( html )
			{
				var d = document.createElement('div');
				d.innerHTML = html;
				document.body.appendChild(d);
				vz_included_elements.push(d);
				return false;
			}
